<body>
    <div class="container">
        <h1>Gráfica Interactiva</h1>
        
        <div class="controls">
            <button onclick="zoomIn()">🔍+ Zoom</button>
            <button onclick="zoomOut()">🔍- Zoom</button>
            <button onclick="resetZoom()">↺ Restablecer</button>
            <span style="margin-left: 20px; font-size: 14px;">
                Zoom: <span id="zoomLevel">100</span>%
            </span>
        </div>
        
        <div id="imageContainer">
            <img id="grafica" src="A1__E_0_02_png.png" alt="Gráfica">
        </div>
    </div>

    <script>
        let zoomLevel = 1;
        let isDragging = false;
        let startX, startY, scrollLeft, scrollTop;
        
        const img = document.getElementById('grafica');
        const container = document.getElementById('imageContainer');
        const zoomDisplay = document.getElementById('zoomLevel');

        function updateZoom() {
            img.style.transform = `scale(${zoomLevel})`;
            img.style.transformOrigin = 'center';
            zoomDisplay.textContent = Math.round(zoomLevel * 100);
        }

        function zoomIn() {
            zoomLevel *= 1.2;
            updateZoom();
        }

        function zoomOut() {
            zoomLevel /= 1.2;
            if (zoomLevel < 1) zoomLevel = 1;
            updateZoom();
        }

        function resetZoom() {
            zoomLevel = 1;
            updateZoom();
        }

        // Zoom con rueda del mouse
        container.addEventListener('wheel', (e) => {
            e.preventDefault();
            if (e.deltaY < 0) zoomIn();
            else zoomOut();
        });

        // Arrastrar para mover
        img.addEventListener('mousedown', (e) => {
            isDragging = true;
            startX = e.pageX;
            startY = e.pageY;
            scrollLeft = container.scrollLeft;
            scrollTop = container.scrollTop;
        });

        document.addEventListener('mousemove', (e) => {
            if (!isDragging) return;
            const x = e.pageX - startX;
            const y = e.pageY - startY;
            container.scrollLeft = scrollLeft - x;
            container.scrollTop = scrollTop - y;
        });

        document.addEventListener('mouseup', () => {
            isDragging = false;
        });
    </script>
